/* arithmetic operation */
{
  int a;
  int b;
  int c;
  int res;
  a = 7;
  b = 15;
  c = 2;
  res = a + b;
  write res;
  res = b - a;
  write res;
  res = b / a;
  write res;
  res = a * b;
  write res;
  res = b % a;
  write res;
  res = a + b * c;
  write res;
  res = (a + b) * c;
  write res;
  res = (a * b) / c;
  write res;
  res = a * (b / c);
  write res;
}
